#install virtual env
pip install virtualenv

#create virtual env
virtualenv myvenv #where myvenv is specifified virtual env name
NB: Ensure both your virtual env and project directory are directly under same parent directory

#to start/activate virtual virtual env on windows
myvenv\scripts\activate

#to stop/deactivate virtual env on windows
enter "deactivate" on cmd

python manage.py startapp appName
